$.views.tags({

    // Tag to reverse-sort an array

    sort: function( array ){
        var ret = "";
        if ( this.props.reverse ) {
            // Render in reverse order
            for ( var i = array.length; i; i-- ) {
                ret += this.renderContent( array[ i - 1 ] );
            }
        } else {
            // Render in original order
            ret += this.renderContent( array );
        }
        return ret;
    }

});
